/*********هذا البرنامج يقوم بعمل الاعدادات للاردوينو*********/

#define LED1 13 //red
#define LED2 12 //yellow
#define LED3 11 //green
#define GREENTIME 10000

void setup()
{
    
    pinMode(LED1, OUTPUT);
    pinMode(LED2, OUTPUT);
    pinMode(LED3, OUTPUT);
}
void loop()
{
  digitalWrite(LED3, HIGH);
    delay(GREENTIME);
    digitalWrite(LED3, LOW);
    delay(300);
    for(int i = 0 ; i < 3; i++)
    {
      digitalWrite(LED3, HIGH); 
      delay(500);
      digitalWrite(LED3, LOW);
      delay(500);
    }
    digitalWrite(LED2, HIGH);
    delay(2000);
    digitalWrite(LED2, LOW);
    delay(300);
    digitalWrite(LED1, HIGH);
    delay(5000);
    digitalWrite(LED1, LOW);
    delay(300);
}
